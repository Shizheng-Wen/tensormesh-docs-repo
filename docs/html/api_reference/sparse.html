Examples

import torch
from tensormesh.sparse import SparseMatrix

# Create a sparse matrix from COO format
edata = torch.tensor([1.0, 2.0, 3.0])
row = torch.tensor([0, 1, 2])
col = torch.tensor([1, 2, 0])
shape = (3, 3)
A = SparseMatrix(edata, row, col, shape)

# Matrix-vector multiplication
x = torch.tensor([1.0, 2.0, 3.0])
y = A @ x

# Matrix-matrix multiplication
B = torch.randn(3, 2)
C = A @ B

# Convert to dense
A_dense = A.to_dense()

# Solve linear system Ax = b
b = torch.tensor([1.0, 2.0, 3.0])
x = A.solve(b)

# Create from scipy sparse matrix
from scipy.sparse import coo_matrix
scipy_mat = coo_matrix(([1,2,3], ([0,1,2], [1,2,0])), shape=(3,3))
A = SparseMatrix.from_scipy(scipy_mat)
